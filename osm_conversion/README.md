	#+begin_src

“highway in "Saint Mandé"”
*/

// fetch area “Saint Mandé” to search in
{{geocodeArea:Saint Mandé}}->.searchArea;
// gather results
(
// query part for: “highway”
way["highway"](area.searchArea);
);
// print results
out body;
>;
